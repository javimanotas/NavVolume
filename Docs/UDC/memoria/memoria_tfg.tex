% Comando auxiliar para marcar figuras o diagramas pendientes de incluir.
% El primer argumento describe el contenido que debe ilustrar la figura,
% el segundo es la leyenda y el tercero la etiqueta para las referencias.
\newcommand{\pendientefig}[3]{%
  \begin{figure}[htp]
    \centering
    \fbox{%
      \parbox[c][3.6cm][c]{0.85\textwidth}{%
        \centering\itshape [Figura pendiente de incluir]\\[6pt]%
        \normalfont\small #1%
      }%
    }%
    \caption{#2}%
    \label{#3}%
  \end{figure}%
}
